# 白山羊备忘录
一个功能简单的会话备忘录的小程序
### 目录结构

    +-- assets          //静态文件夹
    |   +-- font        //字体文件
    |   	+-- iconfont.eot
    |	+-- iconfont.svg
    |	+-- iconfont.ttf
    |	+-- iconfont.woff  
    |   +-- images
    |	+-- share.jpg
    +-- pages		      //页面
    |   +-- add		      //添加备忘录
    |       +-- add.js
    |	+-- add.json 
    |	+-- add.wxml
    |	+-- add.wxss
    |   +-- edit		   //编辑备忘录
    |       +-- edit.js
    |	+-- edit.json 
    |	+-- edit.wxml
    |	+-- edit.wxss
    | +-- index	     //首页
    |       +-- index.js
    |	+-- index.json 
    |	+-- index.wxml
    |	+-- index.wxss
    | +-- logs		  //日志
    |       +-- logs.js
    |	+-- logs.json 
    |	+-- logs.wxml
    |	+-- logs.wxss
    +-- utils		    //公用js
    |   +-- shareData.js    //分享短句
    |   +-- util.js
    +-- app.js
    +-- app.json
    +-- app.wxss
    +-- project.config.json
